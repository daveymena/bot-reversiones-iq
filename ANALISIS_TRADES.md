### **PRIORIDAD MEDIA** 🟡

#### **4. Detector de Cambio de Momentum**
```python
def detect_momentum_shift(recent_trades):
    """
    Detecta si el mercado cambió de dirección
    """
    if len(recent_trades) < 3:
        return False
    
    last_3_directions = [t['direction'] for t in recent_trades[-3:]]
    
    # Si los últimos 3 fueron en la misma dirección
    # y ahora queremos operar en dirección contraria
    # → posible cambio de momentum
    if len(set(last_3_directions)) == 1:
        return True
    
    return False
```

```python
def validate_zone_still_valid(zone, market_data):
    """
    Verifica que la zona no haya sido rota recientemente
    """
    recent_candles = market_data.get_candles(asset, 60, 5)
    
    for candle in recent_candles:
        if zone.zone_type == "resistance":
            if candle['close'] > zone.level * 1.002:
                return False  # Zona rota
        else:
            if candle['close'] < zone.level * 0.998:
                return False  # Zona rota
    
    return True
```
